SELECT
    'ŒûU' AS SYORI_KBN
   ,TORIS_CODE_S
   ,TORIF_CODE_S
   ,BAITAI_CODE_S
   ,SYUBETU_T
   ,FURI_DATE_S
   ,SYORI_KEN_S
   ,SYORI_KIN_S
   ,UKETUKE_FLG_S
   ,SYOUGOU_FLG_S
   ,TOUROKU_FLG_S
   ,DECODE(TOUROKU_FLG_S,'0','1','0') AS MISYOUGOU_FLG
   ,ITAKU_NNAME_T
   ,TKIN_NO_T
   ,TSIT_NO_T
   ,ITAKU_CODE_T 
FROM
    SCHMAST_VIEW
   ,TORIMAST
WHERE
        TORIS_CODE_T = TORIS_CODE_S
    AND TORIF_CODE_T = TORIF_CODE_S
    AND UKETUKE_FLG_S = '1'
    AND SYORI_KEN_S   > 0
    AND (UKETUKE_TIME_STAMP_S Like '{1}%'
    OR   UKETUKE_DATE_S          = '{1}')
ORDER BY 
    BAITAI_CODE_S
   ,FURI_DATE_S
   ,TORIS_CODE_S
   ,TORIF_CODE_S
