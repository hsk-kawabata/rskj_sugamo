SELECT
  E.FSYORI_KBN_ME
  , E.TORIS_CODE_ME
  , E.TORIF_CODE_ME
  , V.ITAKU_NNAME_T
  , E.BAITAI_CODE_ME
  , E.SYUBETU_CODE_ME
  , E.FURI_DATE_ME
  , E.SYORI_KEN_ME
  , E.SYORI_KIN_ME
  , E.CREATE_DATE_ME
  , E.UPDATE_DATE_ME
FROM
  MEDIA_ENTRY_TBL E
  , ( 
    SELECT
      FSYORI_KBN_T
      , TORIS_CODE_T
      , TORIF_CODE_T
      , ITAKU_NNAME_T
      , ITAKU_KNAME_T 
    FROM
      TORIMAST_VIEW
  ) V 
WHERE
  E.FSYORI_KBN_ME = V.FSYORI_KBN_T 
  AND E.TORIS_CODE_ME = V.TORIS_CODE_T 
  AND E.TORIF_CODE_ME = V.TORIF_CODE_T 
  AND E.FURI_DATE_ME = '{1}' 
  AND E.DELETE_FLG_ME = '0' 
  AND E.FSYORI_KBN_ME = '1' 
Order By
  E.TORIS_CODE_ME
  , E.TORIF_CODE_ME
